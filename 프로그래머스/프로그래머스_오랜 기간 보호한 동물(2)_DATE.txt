-- 입양을 간 동물 선택
SELECT OUTS.ANIMAL_ID, OUTS.NAME
FROM ANIMAL_OUTS OUTS
LEFT JOIN ANIMAL_INS INS ON OUTS.ANIMAL_ID = INS.ANIMAL_ID
WHERE INS.DATETIME IS NOT NULL
ORDER BY (OUTS.DATETIME - INS.DATETIME) DESC LIMIT 2
